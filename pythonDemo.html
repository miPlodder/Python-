#class in python

class Employee:
    
    #global variable
    count = 0
    
    def __init__(self, name, salary):
        Employee.count += 1
        self.name = name
        self.salary = salary
        
    def toString(self):
        print("Name : %s\nSalary : %d"%(self.name,self.salary))
        
    def getCount(self):
        print("Total Employee : %d"%(Employee.count))

        
emp1 = Employee("saksham",1000)
emp1.toString()
emp1.getCount()

emp2 = Employee("rahul",2000)
emp2.toString()
emp2.getCount()
